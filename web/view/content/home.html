Add logic in controller/page/mypage.js:

module.exports = function(Candy) {
  Candy.View.skeleton('main')
  Candy.View.set({
    header: 'main',
    content: 'mypage',
    footer: 'main'
  })
}

Customize the look in public/assets/css/style.css. Add your own CSS or use your favorite framework.
